class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int ret = 0;
        if(prices.size() < 1)
            return ret;
        int minbefore = prices[0];
        for(int i=1;i<prices.size();i++){
            int profit = prices[i]-minbefore;
            if(profit >= ret)
                ret = profit;
            if(prices[i]<=minbefore)
                minbefore = prices[i];
        }
        return ret;
    }
};
